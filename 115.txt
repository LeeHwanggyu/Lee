# 첫 페이지 데이터 가져오기 및 CSV 파일 생성
aws connect search-quick-connects --instance-id 7c031296-baf2-4146-b0b7-d3e6e01797fd \
--query "QuickConnects[?QuickConnectConfig.QuickConnectType=='USER'].{Id:QuickConnectId, Name:Name, QuickConnectType:QuickConnectConfig.QuickConnectType, UserId:QuickConnectConfig.UserConfig.UserId, ContactFlowId:QuickConnectConfig.UserConfig.ContactFlowId}" \
--max-results 100 --output json > quickconnects_stg.json

# JSON 데이터를 CSV로 변환하여 파일 생성
jq -r '.[] | [.Id, .Name, .QuickConnectType, .UserId, .ContactFlowId] | @csv' quickconnects_stg.json > quickconnects.csv

# NextToken 추출
NEXT_TOKEN=$(jq -r '.NextToken' quickconnects_stg.json)

# 다음 페이지가 있을 경우 계속해서 데이터를 가져오고 CSV 파일에 추가
while [ "$NEXT_TOKEN" != "null" ] && [ -n "$NEXT_TOKEN" ]; do
    sleep 2  # 지연 추가 (2초)
    
    # 다음 페이지 데이터 가져오기
    aws connect search-quick-connects --instance-id 7c031296-baf2-4146-b0b7-d3e6e01797fd \
    --query "QuickConnects[?QuickConnectConfig.QuickConnectType=='USER'].{Id:QuickConnectId, Name:Name, QuickConnectType:QuickConnectConfig.QuickConnectType, UserId:QuickConnectConfig.UserConfig.UserId, ContactFlowId:QuickConnectConfig.UserConfig.ContactFlowId}" \
    --max-results 100 --starting-token "$NEXT_TOKEN" --output json > quickconnects_stg.json
    
    # JSON 데이터를 CSV로 변환하여 파일에 추가
    jq -r '.[] | [.Id, .Name, .QuickConnectType, .UserId, .ContactFlowId] | @csv' quickconnects_stg.json >> quickconnects.csv
    
    # NextToken 추출
    NEXT_TOKEN=$(jq -r '.NextToken' quickconnects_stg.json | tail -n 1)
done
